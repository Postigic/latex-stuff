\begin{claim}
    Let $a_n$ be the number of length $n$ ternary strings that contain two consecutive digits that are the same. 
    Then $a_n = 3^{n-1} + 2a_{n-1}$ for $n \ge 2$ with $a_0 = 0$ and $a_1 = 0$.
\end{claim}

\begin{proof}
    Let $b_n$ denote the number of length $n$ ternary strings that contain no two consecutive equal digits. 
    Every such string counted by $b_n$ can be extended to a string counted by $b_{n+1}$ in exactly two ways by 
    appending either of the two digits different from the original string's final digit. Conversely, removing 
    the final digit from any string counted by $b_{n+1}$ yields a unique string counted by $b_n$. Hence
    \[
    b_{n+1} = 2b_n \quad n \ge 1
    \]
    Since $b_1 = 3$, it follows that
    \[
    b_n = 3 \cdot 2^{n-1} \quad n \ge 1
    \]

    \medskip

    Now, every length $n$ ternary string either contains two consecutive equal digits or does not. Therefore,
    \[
    a_n + b_n = 3^n
    \]
    Using the recurrence for $b_n$, we get
    \[
    \begin{aligned}
    a_{n+1} &= 3^{n+1} - b_{n+1} \\
    &= 3(3^n) - 2b_n \\
    &= 3(3^n) - 2(3^n - a_n) \\
    &= 3^n + 2a_n
    \end{aligned}
    \]
    Finally, $a_0 = 0$ and $a_1 = 0$ since the empty string and single-digit strings have no consecutive 
    digits. Replacing $n$ with $n - 1$ gives the equivalent recurrence
    \[
    a_n = 3^{n-1} + 2a_{n-1}
    \]
    where $n \ge 2$ with $a_0 = 0$ and $a_1 = 0$.
    
    \medskip

    Hence $a_n = 3^{n-1} + 2a_{n-1}$ for $n \ge 2$ with $a_0 = 0$ and $a_1 = 0$.
\end{proof}

\begin{claim}
    The generating function of the sequence $a_0, a_1, \dots$ is
    \[
    \frac{-x}{1 - 2x} + \frac{x}{(1 - 3x)(1 - 2x)}.
    \]
\end{claim}

\begin{proof}
    Define
    \[
    A(x) = \sum_{n=0}^{\infty}a_nx^n
    \]
    From Claim 1, we know that the recurrence formula for the sequence is
    \[
    a_n = 3^{n-1} + 2a_{n-1}
    \]
    for $n \ge 2$ with $a_0 = 0$ and $a_1 = 0$. Hence we derive a closed form for $A(x)$ as follows:
    \[
    \begin{aligned}
    a_n &= 3^{n-1} + 2a_{n-1} \\
    \sum_{n=2}^{\infty}a_nx^n &= \sum_{n=2}^{\infty}3^{n-1}x^n + 2\sum_{n=2}^{\infty}a_{n-1}x^n \\
    A(x) - a_0 - a_1x &= x\sum_{k=1}^{\infty}(3x)^k + 2x\sum_{k=1}^{\infty}a_kx^k \quad (k = n - 1) \\
    A(x) &= x\left(\frac{3x}{1 - 3x}\right) + 2x(A(x) - a_0) \\
    A(x) &= \frac{3x^2}{1 - 3x} + 2xA(x) \\
    A(x) - 2xA(x) &= \frac{3x^2}{1 - 3x} \\
    A(x)(1 - 2x) &= \frac{3x^2}{1 - 3x} \\
    A(x) &= \frac{3x^2}{(1 - 3x)(1 - 2x)}
    \end{aligned}
    \]

    \medskip

    It remains to verify that this expression is algebraically equivalent to the form stated in the claim:
    \[
    \begin{aligned}
    \frac{-x}{1 - 2x} + \frac{x}{(1 - 3x)(1 - 2x)} &= \frac{-x(1 - 3x) + x}{(1 - 3x)(1 - 2x)} \\
    &= \frac{3x^2}{(1 - 3x)(1 - 2x)}
    \end{aligned}
    \]

    \medskip

    Hence
    \[
    \frac{-x}{1 - 2x} + \frac{x}{(1 - 3x)(1 - 2x)}
    \]
    is a closed form for the generating function of the sequence $a_0, a_1, \dots$
\end{proof}

\begin{claim}
    \[
    \frac{1}{(1 - 2x)(1 - 3x)} = -\frac{2}{1 - 2x} + \frac{3}{1 - 3x}
    \]
\end{claim}

\begin{proof}
    By partial fraction decomposition,
    \[
    \begin{aligned}
    \frac{1}{(1 - 2x)(1 - 3x)} &= \frac{A}{1 - 2x} + \frac{B}{1 - 3x} \\
    1 &= A(1 - 3x) + B(1 - 2x) \\
    \text{Let } x &= \frac{1}{3} \\
    1 &= A\left(1 - 3\left(\frac{1}{3}\right)\right) + B\left(1 - 2\left(\frac{1}{3}\right)\right) \\
    \frac{B}{3} &= 1 \\
    B &= 3 \\
    \text{Let } x &= 1 \\
    1 &= A(1 - 3(1)) + 3(1 - 2(1)) \\
    1 &= -2A - 3 \\
    -2A &= 4 \\
    A &= -2 \\
    \frac{1}{(1 - 2x)(1 - 3x)} &= -\frac{2}{1 - 2x} + \frac{3}{1 - 3x}
    \end{aligned}
    \]

    \medskip

    Hence
    \[
    \frac{1}{(1 - 2x)(1 - 3x)} = -\frac{2}{1 - 2x} + \frac{3}{1 - 3x}
    \]
\end{proof}

\begin{claim}
    The closed form for the $n$th term in the sequence is $a_n = 3^n - 3\cdot 2^{n-1}$ for $n \ge 1$ with 
    $a_0 = 0$.
\end{claim}

\begin{proof}
    From Claim 1, we know that there are $3^n$ length $n$ ternary strings, the number of length $n$ 
    ternary strings which do not contain two consecutive equal digits is $3 \cdot 2^{n-1}$ for $n \ge 1$, 
    and $a_0 = 0$. Therefore, for $n \ge 1$, the closed form for the $n$th term in the sequence is
    \[
    3^n - 3 \cdot 2^{n-1}
    \]

    \medskip

    Hence the closed form for the $n$th term in the sequence is $a_n = 3^n - 3 \cdot 2^{n-1}$ for $n \ge 1$ 
    with $a_0 = 0$.
\end{proof}
